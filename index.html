Selenium Client For NodeJS

Install (requires NPM):

npm install soda

Create `mytest.js`:

var soda = require('soda');

var browser = soda.createSauceClient({
  'url': 'http://yourwebsite/'
, 'username': ''
, 'access-key': ''
, 'os': 'Windows 2003'
, 'browser': 'googlechrome'
, 'max-duration': 300
});

browser
  .chain
  .session()
  .setTimeout(5000)
  .open('/')
  .waitForPageToLoad(10000)
  .click('link=Login')
  .waitForPageToLoad(10000)
  .waitForElementPresent(name=email)
  .type('name=email', 'invalid@example.com')
  .type('name=password', 'invalid')
  .click('css=input.btn-submit')
  .waitForTextPresent('Please check your email / password')
  .assertTextPresent('Please check your email / password')
  .testComplete()
  .end(function(err){
    if (err) throw err;
    console.log('Passed!');
  });

Run it!

node mytest.js
